03 WordPress Functions

Functions.php

                
            
             #Adjust if needed

              <?php
                function remove_content_more_link_id($content) {
                    global $post;
                    $content = str_replace('#more-' . $post->ID, " ", $content);
                    return $content;
              }
              
              add_filter('the_content', 'remove_content_more_link_id');
              ?>
